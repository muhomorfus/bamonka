\begin{definitions}
	\definition{База данных}{совокупность данных, организованных по определенным правилам, предусматривающим общие принципы описания, хранения и манипулирования данных, независимая от прикладных программ~\cite{карпова2021базы}.}
	\definition{Кластер}{совокупность компьютеров, выполняющих общую задачу, и ведущую себя в рамках нее, как единое целое~\cite{тестова2023организация}.}
	\definition{Запрос}{текст, составленный с использорванием заданного синтаксиса, позволяющий получить информацию из базы данных~\cite{микаелян2021анализ}.}
	\definition{Транзакция}{набор операций, выполняющихся атомарно, как единое целое~\cite{фомин2021проблематика}.}
\end{definitions}
